    t.async = true;
    t.id    = 'weatherglass-tracker';
    t.setAttribute('data-site-id', 'your unique id here');

Laravel

Github Repository 

    composer require jonahgeorge/weatherglass-laravel
In config/weatherglass.php add your site_id. 

return [
    'site_id' => 'YOUR SITE ID',
];
In app/Http/Kernel.php, add the Weatherglass middleware to your web middleware group. 

protected $middlewareGroups = [
    'web' => [
        \JonahGeorge\Middleware\Weatherglass::class,
    ],
];
